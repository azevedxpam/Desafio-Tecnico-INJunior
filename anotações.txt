function verificarIdades(anosPessoas, anoAtual) {
  let quantPessoas = anosPessoas.length;
  let idadePessoas = [];

  for (let i = 0; i < quantPessoas; i++) {
    let idade = anoAtual - anosPessoas[i];
    idadePessoas[i] = idade;
  }
  //array para armazenar se Ã© maior ou menor de idade
  let arrayVerificador = [];

  for (let i = 0; i < quantPessoas; i++) {
    let verificador;
    if (idadePessoas[i] >= 18) {
      verificador = "maior";
    } else {
      verificador = "menor";
    }

    console.log(`Pessoa ${i + 1}: ${verificador} de idade\n`);
    arrayVerificador[i] = verificador;
  }

  return arrayVerificador;
}

let anos = [1981, 2010, 2005, 2019];
let anoAtual = 2025;
verificarIdades(anos, anoAtual);
